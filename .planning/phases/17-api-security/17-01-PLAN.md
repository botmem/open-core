<objective>
Lock down the API: register JwtAuthGuard globally so all endpoints require authentication by default, mark public endpoints with @Public(), configure CORS to only allow the frontend origin with credentials, and add JWT verification to WebSocket handshake.

Purpose: Without this, any unauthenticated client can access all API endpoints and WebSocket events. This phase makes the secure-by-default posture real.
Output: All endpoints protected, public routes explicitly opted out, CORS locked, WebSocket authenticated.
</objective>

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private reflector: Reflector) { super(); }
  canActivate(context: ExecutionContext) {
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [
      context.getHandler(), context.getClass(),
    ]);
    if (isPublic) return true;
    return super.canActivate(context);
  }
}
